29uint8_t prep_mask(uint8_t p_bit, uint8_t n_bit)
30{
31// Preparation du masque
32 uint8_t mask_val = 0b1;
33 uint8_t mask_temp = 0b1;
34 // Construit n_bit bits a 1
35 for (int i = 0; i < (n_bit - 1); i++)
36 {
37 mask_val = mask_val << 1;
38 mask_val |= mask_temp;
39 }
40 // Aligne le masque sur la position cible dans le byte

44CAN_TG_STATUE set_value(const char* tag_name, uint32_t value)
45{
46 const TagDef* tag = get_tag_def(tag_name);
47 if (!tag) return -3; // Tag non trouve
48
49 const uint8_t n_bits = tag->n_bits;
50 const uint8_t bit_pos_a = tag->bit_pos_a;
51 const uint8_t bit_pos_b = tag->bit_pos_b;
52 const int8_t byte_idx_a = tag->byte_idx_a;
53 const int8_t byte_idx_b = tag->byte_idx_b;
54
55 // Verification du depassement de la valeur selon n_bits
56 if (tag->n_bits < 32 && value >= (1ULL << tag->n_bits))
57 {
58 printf("[ERREUR] Valeur ٪u trop grande pour le tag ٪s (٪u bits)\n", value, tag_name, tag->n_bits);
59 return CAN_TG_ERROR_VALUE_TO_BIG;
60 }
61
62 // Cas des valeurs sur plus de 8 bits: repartition entre partie A et partie B.
63 if(n_bits > 8){
64 uint8_t data_b = (value & 0xFF00) >> 8;
65 uint8_t data_a = (value & 0x00FF);
66 TxData[byte_idx_a] = data_a;
67 // Cas 16 bits: la partie B occupe tout le byte.
68 if(n_bits == 16){TxData[tag->byte_idx_b] = data_b;}
69 else{
70 // Cas 9..15 bits: insertion partielle dans le byte B.
71 // Creation du masque binaire
72 uint8_t mask = prep_mask(bit_pos_b, n_bits - 8);
73 // Alignement de la deuxieme partie
74 data_b = data_b << (8 - bit_pos_b - n_bits);
75 // Mise a jour selective des bits de la partie B.
76 TxData[byte_idx_b] &= ~mask;
77 TxData[byte_idx_b] |= data_b;
78 return CAN_TG_SUCCESS;
79 }
80 }
81 else if(tag->n_bits == 8)
82 {
83 uint8_t data_a = (value & 0x00FF);
84 TxData[tag->byte_idx_a] = data_a;
85 return CAN_TG_SUCCESS;
86 }
87 else{
88 // Valeur <= 8 bits (entre 1 et 7 bits inclusivement)
89 // Creation du masque binaire aligne sur la zone cible
90 uint8_t mask = prep_mask(bit_pos_a, n_bits);
91 uint8_t vrai_val = value;
92 // Alignement de la valeur dans le byte
93 vrai_val = vrai_val << (8 - bit_pos_a - n_bits);
94 TxData[byte_idx_a] &= ~mask;
95 TxData[byte_idx_a] |= vrai_val;
96 }
97 return CAN_TG_SUCCESS;
98}
