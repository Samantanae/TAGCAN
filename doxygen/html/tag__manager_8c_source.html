144CAN_TG_STATUE find_byte_with_enoung_space(uint8_t siz, int8_t* retur_var)
145{
146 // Pour > 8 bits, la 1re recherche se limite a un bloc de 8 bits.
147 if(siz > 8)
148 {
149 siz = 8;
150 }
151 // Pour les tailles impaires: priorite aux bytes a place impaire restante.
152 if(siz%2 > 0)
153 {
154 // Heuristique de placement.
155 for (int8_t i = 0; i < N_BYTES; i++)
156 {
157 {
158 if(((8 - g_bytes_used[i])%2 > 0) && ((8 - g_bytes_used[i]) >= siz))
159 {
160 *retur_var=i;
161 return CAN_TG_SUCCESS;
162 }
163 }
164 }
165 }
166 // Sinon, recherche sequentielle standard.
167 for (int8_t i = 0; i < N_BYTES; i++)
168 {
169 if((8 - g_bytes_used[i]) >= siz)
170 {
171 *retur_var=i;
172 return CAN_TG_SUCCESS;
173 }
174 }
175 return CAN_TG_ERROR_NOT_ENOUNG_SPACE_;
176}
