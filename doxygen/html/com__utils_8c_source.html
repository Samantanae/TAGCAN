84CAN_FilterTypeDef* CAN_setup_Filter(int32_t minimum_id, int32_t maximum_id, int filterBank){
85 // understanding this https://youtu.be/JfWlIY0zAIc
86 CAN_FilterTypeDef canfilterconfig;
87 canfilterconfig.FilterActivation = CAN_FILTER_ENABLE;
88 canfilterconfig.FilterBank = filterBank; // Assign and Select Filter Banks
89 canfilterconfig.FilterFIFOAssignment = CAN_FILTER_FIFO0;
90 canfilterconfig.FilterIdHigh = maximum_id <<5;
91 canfilterconfig.FilterMaskIdHigh = maximum_id << 5;
92 canfilterconfig.FilterIdLow = minimum_id;
93 canfilterconfig.FilterMaskIdLow = minimum_id;
94 canfilterconfig.FilterMode = CAN_FILTERMODE_IDMASK;
95 canfilterconfig.FilterScale = CAN_FILTERSCALE_32BIT; // 16 ou 32 bits register
96 canfilterconfig.SlaveStartFilterBank = 20; // how many filters to assign to the CAN1 (master can)
97
98 HAL_CAN_ConfigFilter(&hcan, &canfilterconfig);
99 return &canfilterconfig;
100}
